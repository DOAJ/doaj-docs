31 def validate_file(self, file_handle): 

32 # first try to parse the file 

33 try: 

34 doc = etree.parse(file_handle) 

35 except etree.XMLSyntaxError as e: # although the treatment is the same, pulling this out so we remember what the primary kind of exception should be 

36 raise CrosswalkException(message="Unable to parse XML file", inner=e) 

37 except UnicodeDecodeError as e: 

38 msg = 'Text decode failed, expected utf-8 encoded XML.' 

39 raise CrosswalkException(message='Unable to parse XML file', inner=e, inner_message=msg) 

40 except Exception as e: 

41 raise CrosswalkException(message="Unable to parse XML file", inner=e) 

42 

43 # then pass the doc to the validator 

44 valid = self.validate(doc) 

45 

46 if not valid: 

47 msg = "Validation message from schema '{x}': {y}\n".format(x=CrossrefXWalk.format_name, 

48 y=self.validation_log) 

49 raise CrosswalkException(message="Unable to validate document with identified schema", inner_message=msg) 

50 

51 return doc

53 def validate(self, doc): 

54 valid = self.schema.validate(doc) 

55 if not valid: 

56 el = self.schema.error_log.__repr__() 

57 # strip the filename, as we don't want to leak the path to the UI 

58 rx = "[\da-f]{32}.xml:(.*)" 

59 match = re.search(rx, el) 

60 if match is not None: 

61 el = match.group(1) 

62 self.validation_log = el 

63 return valid

69 def crosswalk_doc(self, doc): 

70 # go through the records in the doc and crosswalk each one individually 

71 articles = [] 

72 root = doc.getroot() 

73 body = root.find("x:body", NS) 

74 journals = body.findall("x:journal", NS) 

75 if journals is not None: 

76 for journal in journals: 

77 arts = journal.findall("x:journal_article", NS) 

78 for record in arts: 

79 article = self.crosswalk_article(record, journal) 

80 articles.append(article) 

81 

82 return articles
