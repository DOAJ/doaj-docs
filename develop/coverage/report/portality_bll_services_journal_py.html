18 def journal_2_application(self, journal, account=None, keep_editors=False): 

19 """ 

20 Function to convert a given journal into an application object. 

21 

22 Provide the journal, and it will be converted 

23 in-memory to the application object (currently a Suggestion). The new application 

24 WILL NOT be saved by this method. 

25 

26 If an account is provided, this will validate that the account holder is 

27 allowed to make this conversion 

28 

29 :param journal: a journal to convert 

30 :param account: an account doing the action - optional, if specified the application will only be created if the account is allowed to 

31 :return: Suggestion object 

32 """ 

33 

34 # first validate the incoming arguments to ensure that we've got the right thing 

35 argvalidate("journal_2_application", [ 

36 {"arg": journal, "instance" : models.Journal, "allow_none" : False, "arg_name" : "journal"}, 

37 {"arg" : account, "instance" : models.Account, "arg_name" : "account"} 

38 ], exceptions.ArgumentException) 

39 

40 if app.logger.isEnabledFor(logging.DEBUG): app.logger.debug("Entering journal_2_application") 

41 

42 authService = DOAJ.authorisationService() 

43 

44 # if an account is specified, check that it is allowed to perform this action 

45 if account is not None: 

46 try: 

47 authService.can_create_update_request(account, journal) # throws exception if not allowed 

48 except exceptions.AuthoriseException as e: 

49 msg = "Account {x} is not permitted to create an update request on journal {y}".format(x=account.id, y=journal.id) 

50 app.logger.info(msg) 

51 e.args += (msg,) 

52 raise 

53 

54 # copy all the relevant information from the journal to the application 

55 bj = journal.bibjson() 

56 notes = journal.notes 

57 

58 application = models.Suggestion() 

59 application.set_application_status(constants.APPLICATION_STATUS_UPDATE_REQUEST) 

60 application.set_current_journal(journal.id) 

61 if keep_editors is True: 

62 if journal.editor is not None: 

63 application.set_editor(journal.editor) 

64 if journal.editor_group is not None: 

65 application.set_editor_group(journal.editor_group) 

66 for n in notes: 

67 # NOTE: we keep the same id for notes between journal and application, since ids only matter within 

68 # the scope of a record there are no id clashes, and at the same time it may be useful in future to 

69 # check the origin of some journal notes by comparing ids to application notes. 

70 application.add_note(n.get("note"), n.get("date"), n.get("id")) 

71 application.set_owner(journal.owner) 

72 application.set_seal(journal.has_seal()) 

73 application.set_bibjson(bj) 

74 application.date_applied = dates.now() 

75 

76 if app.logger.isEnabledFor(logging.DEBUG): app.logger.debug("Completed journal_2_application; return application object") 

77 return application

79 def journal(self, journal_id, lock_journal=False, lock_account=None, lock_timeout=None): 

80 """ 

81 Function to retrieve a journal by its id, and to optionally lock the resource 

82 

83 May raise a Locked exception, if a lock is requested but can't be obtained. 

84 

85 :param journal_id: the id of the journal 

86 :param: lock_journal: should we lock the resource on retrieval 

87 :param: lock_account: which account is doing the locking? Must be present if lock_journal=True 

88 :param: lock_timeout: how long to lock the resource for. May be none, in which case it will default 

89 :return: Tuple of (Journal Object, Lock Object) 

90 """ 

91 # first validate the incoming arguments to ensure that we've got the right thing 

92 argvalidate("journal", [ 

93 {"arg": journal_id, "allow_none" : False, "arg_name" : "journal_id"}, 

94 {"arg": lock_journal, "instance" : bool, "allow_none" : False, "arg_name" : "lock_journal"}, 

95 {"arg": lock_account, "instance" : models.Account, "allow_none" : True, "arg_name" : "lock_account"}, 

96 {"arg": lock_timeout, "instance" : int, "allow_none" : True, "arg_name" : "lock_timeout"} 

97 ], exceptions.ArgumentException) 

98 

99 # retrieve the journal 

100 journal = models.Journal.pull(journal_id) 

101 

102 # if we've retrieved the journal, and a lock is requested, request it 

103 the_lock = None 

104 if journal is not None and lock_journal: 

105 if lock_account is not None: 

106 the_lock = lock.lock(constants.LOCK_JOURNAL, journal_id, lock_account.id, lock_timeout) 

107 else: 

108 raise exceptions.ArgumentException("If you specify lock_journal on journal retrieval, you must also provide lock_account") 

109 

110 return journal, the_lock

132 def _make_journals_csv(file_object): 

133 """ 

134 Make a CSV file of information for all journals. 

135 :param file_object: a utf8 encoded file object. 

136 """ 

137 

138 cols = {} 

139 for j in models.Journal.all_in_doaj(page_size=100000): # 10x how many journals we have right now 

140 assert isinstance(j, models.Journal) # for pycharm type inspection 

141 bj = j.bibjson() 

142 issn = bj.get_one_identifier(idtype=bj.P_ISSN) 

143 if issn is None: 

144 issn = bj.get_one_identifier(idtype=bj.E_ISSN) 

145 if issn is None: 

146 continue 

147 

148 kvs = Journal2QuestionXwalk.journal2question(j) 

149 meta_kvs = _get_doaj_meta_kvs(j) 

150 article_kvs = _get_article_kvs(j) 

151 cols[issn] = kvs + meta_kvs + article_kvs 

152 

153 # Get the toc URL separately from the meta kvs because it needs to be inserted earlier in the CSV 

154 toc_kv = _get_doaj_toc_kv(j) 

155 cols[issn].insert(2, toc_kv) 

156 

157 issns = cols.keys() 

158 

159 csvwriter = csv.writer(file_object) 

160 qs = None 

161 for i in sorted(issns): 

162 if qs is None: 

163 qs = [q for q, _ in cols[i]] 

164 csvwriter.writerow(qs) 

165 vs = [v for _, v in cols[i]] 

166 csvwriter.writerow(vs)
