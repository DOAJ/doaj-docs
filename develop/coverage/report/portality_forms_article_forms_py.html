398 def _render_field(self, field, **kwargs): 

399 # interesting arguments from keywords 

400 extra_input_fields = kwargs.get("extra_input_fields") 

401 q_num = kwargs.pop("q_num", None) 

402 maximise_width = kwargs.pop("maximise_width", False) 

403 clazz = kwargs.get("class", "") 

404 label_width = kwargs.get("label_width", 3) 

405 field_width = 12 - label_width 

406 field_width = str(kwargs.get("field_width", field_width)) 

407 if label_width > 0: 

408 label_width = str(label_width) 

409 

410 if field.type == 'CSRFTokenField' and not field.value: 

411 return "" 

412 

413 frag = "" 

414 

415 # If this is the kind of field that requires a label, give it one 

416 if field.type not in ['SubmitField', 'HiddenField', 'CSRFTokenField']: 

417 if q_num is not None: 

418 frag += '<a class="animated" name="' + q_num + '"></a>' 

419 if label_width != 0: 

420 frag += '<label class="control-label col-md-' + label_width + '" for="' + field.short_name + '">'

425 frag += '&nbsp;<span class="red">*</span>' 

426 frag += "</label>"

428 # determine if this is a checkbox 

429 is_checkbox = False 

430 if (field.type == "SelectMultipleField" 

431 and field.option_widget.__class__.__name__ == 'CheckboxInput' 

432 and field.widget.__class__.__name__ == 'ListWidget'):

437 extra_class += " checkboxes"

439 frag += '<div class="col-md-' + field_width + ' ' + extra_class + '">'

442 frag += self._render_radio(subfield, **kwargs)

444 frag += '<ul id="' + field.short_name + '">'

448 else: 

449 if maximise_width: 

450 clazz += " col-xs-12" 

451 kwargs["class"] = clazz

457 frag += field(**render_args) # FIXME: this is probably going to do some weird stuff

463 frag += "</ul></div>" 

464 

465 if field.description: 

466 frag += '<p class="help-block">' + field.description + '</p>' 

467 

468 frag += "</div>"
