1""" 

2S.E. 2021-11-18 

3Script for issue 3155 - https://github.com/DOAJ/doajPM/issues/3155 

4 

5* Read in provided CSV 

6* Extract journal IDs from doaj ToC URL 

7* Get owner account for journal 

8* Add new columns to CSV for account, email address. 

9""" 

10 

11import csv 

12 

13from portality.lib import dates 

14from portality.models import Journal 

15 

16if __name__ == '__main__': 

17 print('Starting {0}.'.format(dates.now())) 

18 

19 import argparse 

20 parser = argparse.ArgumentParser() 

21 parser.add_argument('-c', '--csv', help='csv containing the journals to add contact details to', required=True) 

22 parser.add_argument('-o', '--out', help='path to write the output to', required=True) 

23 args = parser.parse_args() 

24 

25 with open(args.csv, 'r', encoding='utf-8') as f: 

26 reader = csv.DictReader(f) 

27 with open(args.out, 'w', encoding='utf-8') as g: 

28 new_headers = list(reader.fieldnames) + ['Account ID', 'Account Email'] 

29 writer = csv.DictWriter(g, new_headers) 

30 writer.writeheader() 

31 

32 for row in reader: 

33 j_id = row['URL in DOAJ'].split('/').pop() 

34 # Get the Journal and its owner details via the ID 

35 try: 

36 j = Journal.pull(j_id) 

37 o = j.owner_account 

38 row.update({'Account ID': o.id or '', 'Account Email': o.email or ''}) 

39 except AttributeError: 

40 # else we haven't found the owner at all 

41 msg = 'ACCOUNT MISSING: ' + j.owner if j else "JOURNAL MISSING" 

42 row.update({'Account ID': msg, 'Account Email': ''}) 

43 

44 writer.writerow(row) 

45 

46 print('Finished {0}.'.format(dates.now()))
