13SCHEMA_TEMPLATE = '''\ 

14<xsd:schema attributeFormDefault="unqualified" elementFormDefault="qualified" targetNamespace="http://www.doaj.org/schemas/iso_639-2b/{schema_version}" xmlns:xsd="http://www.w3.org/2001/XMLSchema"> 

15 <xsd:annotation> 

16 <xsd:documentation>Codes for the representation of names of languages from the International Organization for Standardization (ISO) 639-2/B (bibliographic codes).</xsd:documentation> 

17 </xsd:annotation> 

18 <xsd:simpleType name="LanguageCodeType"> 

19 <xsd:annotation> 

20 <xsd:documentation>A code list that enumerates languages.</xsd:documentation> 

21 </xsd:annotation> 

22 <xsd:restriction base="xsd:token"> 

23 </xsd:restriction> 

24 </xsd:simpleType> 

25</xsd:schema> 

26''' 

27 

28 

29""" 

30Each language entry looks like this, listed within <xsd:restriction>: 

31 

32<xsd:enumeration value="{language_code}"> 

33 <xsd:annotation> 

34 <xsd:documentation>{language_name}</xsd:documentation> 

35 </xsd:annotation> 

36</xsd:enumeration> 

37"""

40def write_lang_schema(out_file, schema_version): 

41 """ Generate an XML file with language data from pycountry 

42 :param out_file 

43 :param schema_version 

44 """ 

45 

46 # Set up our document as parsed XML 

47 parser = etree.XMLParser(remove_blank_text=True) 

48 schema = etree.XML(SCHEMA_TEMPLATE.format(schema_version=schema_version), parser=parser) 

49 schema_tree = etree.ElementTree(schema) 

50 language_list_element = schema.find('.//xsd:restriction', namespaces=schema.nsmap) 

51 language_list_element.text = None # Fixes whitespace failure on pretty_print 

52 

53 # Define the building blocks for a language entry in the Schema 

54 E = ElementMaker(namespace=schema.nsmap[schema.prefix], nsmap=schema.nsmap) 

55 ENUM = E.enumeration 

56 ANNOT = E.annotation 

57 DOCU = E.documentation 

58 

59 # Gather names and 3-char codes (bibliographic preferred) for only the languages with 2-character codes (ISO639-1) 

60 for l in language_options: 

61 try: 

62 _ = l.alpha_2 

63 except AttributeError: 

64 continue # Skip languages without 2-char codes 

65 

66 try: 

67 code = l.bibliographic 

68 except AttributeError: 

69 code = l.alpha_3 # Fallback to alpha_3 when no bibliographic code 

70 

71 # See docstring above for what the XML for this looks like 

72 language_entry = ENUM( 

73 {'value': code}, 

74 ANNOT( 

75 DOCU(l.name) 

76 ) 

77 ) 

78 

79 # Add the language to our document 

80 language_list_element.append(language_entry) 

81 

82 # Write the new XML Schema file 

83 schema_tree.write(out_file, pretty_print=True, encoding='utf-8', xml_declaration=True)

86def compare_lang_schemas(schema_old, schema_new, ofile): 

87 """ Generate a simplified view of the old and new schema, then diff their contents """ 

88 # Parse the XML Schemata for comparison 

89 old_tree = etree.parse(schema_old).getroot() 

90 new_tree = etree.parse(schema_new).getroot() 

91 

92 # Extract the language information from both trees 

93 old_lc = [a.attrib['value'] for a in old_tree.findall('.//xsd:restriction/xsd:enumeration', namespaces=old_tree.nsmap)] 

94 old_ln = [a.text for a in old_tree.findall('.//xsd:restriction/xsd:enumeration/xsd:annotation/xsd:documentation', 

95 namespaces=old_tree.nsmap)] 

96 

97 new_lc = [b.attrib['value'] for b in new_tree.findall('.//xsd:restriction/xsd:enumeration', namespaces=old_tree.nsmap)] 

98 new_ln = [b.text for b in new_tree.findall('.//xsd:restriction/xsd:enumeration/xsd:annotation/xsd:documentation', 

99 namespaces=new_tree.nsmap)] 

100 

101 # List the simplified language entries 

102 old_strlist = ['{0}\t{1}'.format(t[0], t[1]) for t in zip(old_lc, old_ln)] 

103 new_strlist = ['{0}\t{1}'.format(t[0], t[1]) for t in zip(new_lc, new_ln)] 

104 

105 old_file = schema_old.split('/').pop() 

106 new_file = schema_new.split('/').pop() 

107 

108 diff = difflib.HtmlDiff().make_file(old_strlist, new_strlist, fromdesc=old_file, todesc=new_file, context=True) 

109 

110 with open(ofile, 'w') as o: 

111 o.writelines(diff) 

112 

113 print("Diff saved to " + ofile)
