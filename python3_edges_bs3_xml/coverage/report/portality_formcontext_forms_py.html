ISSN_REGEX = re.compile(r'^\d{4}-\d{3}(\d|X|x){1}$') 

ISSN_ERROR = 'An ISSN or EISSN should be 7 or 8 digits long, separated by a dash, e.g. 1234-5678. If it is 7 digits long, it must end with the letter X (e.g. 1234-567X).'
