189@app.template_filter("bytesToFilesize") 

190def bytes_to_filesize(size): 

191 units = ["bytes", "Kb", "Mb", "Gb"] 

192 scale = 0 

193 while size > 1000 and scale < len(units): 

194 size = float(size) / 1000.0 # note that it is no longer 1024 

195 scale += 1 

196 return "{size:.1f}{unit}".format(size=size, unit=units[scale])

199@app.template_filter('utc_timestamp') 

200def utc_timestamp(stamp, string_format="%Y-%m-%dT%H:%M:%SZ"): 

201 """ 

202 Format a local time datetime object to UTC 

203 :param stamp: a datetime object 

204 :param string_format: defaults to "%Y-%m-%dT%H:%M:%SZ", which complies with ISO 8601 

205 :return: the string formatted datetime 

206 """ 

207 local = tzlocal.get_localzone() 

208 ld = local.localize(stamp) 

209 tt = ld.utctimetuple() 

210 utcdt = datetime(tt.tm_year, tt.tm_mon, tt.tm_mday, tt.tm_hour, tt.tm_min, tt.tm_sec, tzinfo=pytz.utc) 

211 return utcdt.strftime(string_format)

233@app.template_filter('form_diff_table_comparison_value') 

234def form_diff_table_comparison_value(val): 

235 """ 

236 Function for converting the given value to a suitable UI value for presentation in the diff table 

237 on the admin forms for update requests. 

238 

239 :param val: the raw value to be converted to a display value 

240 :return: 

241 """ 

242 if val is None: 

243 return "" 

244 if isinstance(val, list) and len(val) == 0: 

245 return "" 

246 

247 if isinstance(val, list): 

248 dvals = [] 

249 for v in val: 

250 dvals.append(form_diff_table_comparison_value(v)) 

251 return ", ".join(dvals) 

252 else: 

253 if val is True or (isinstance(val, str) and val.lower() == "true"): 

254 return "Yes" 

255 elif val is False or (isinstance(val, str) and val.lower() == "false"): 

256 return "No" 

257 return val

260@app.template_filter('form_diff_table_subject_expand') 

261def form_diff_table_subject_expand(val): 

262 """ 

263 Function for expanding one or more subject classifications out to their full terms 

264 

265 :param val: 

266 :return: 

267 """ 

268 if val is None: 

269 return "" 

270 if isinstance(val, list) and len(val) == 0: 

271 return "" 

272 if not isinstance(val, list): 

273 val = [val] 

274 

275 from portality import lcc 

276 

277 results = [] 

278 for v in val: 

279 if v is None or v == "": 

280 continue 

281 expanded = lcc.lcc_index_by_code.get(v) 

282 if expanded is not None: 

283 results.append(expanded + " [code: " + v + "]") 

284 else: 

285 results.append(v) 

286 

287 return ", ".join(results)

301@app.before_request 

302def standard_authentication(): 

303 """Check remote_user on a per-request basis.""" 

304 remote_user = request.headers.get('REMOTE_USER', '') 

305 if remote_user: 

306 user = models.Account.pull(remote_user) 

307 if user: 

308 login_user(user, remember=False) 

309 elif 'api_key' in request.values: 

310 q = models.Account.query(q='api_key:"' + request.values['api_key'] + '"') 

311 if 'hits' in q: 

312 res = q['hits']['hits'] 

313 if len(res) == 1: 

314 user = models.Account.pull(res[0]['_source']['id']) 

315 if user: 

316 login_user(user, remember=False)
