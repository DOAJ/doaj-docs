33# Terms search template. Ensure all queries from OpenURL return publicly visible results with in_doaj : true 

34IN_DOAJ_TERM = {"term": {"admin.in_doaj": True}} 

35TERMS_SEARCH = {"query": {"bool": {"must": [IN_DOAJ_TERM]}}}

38class OpenURLRequest(object): 

39 """ 

40 Based on the fields from ofi/fmt:kev:mtx:journal schema for Journals in OpenURL 1.0 

41 This is the only schema the DOAJ supports. 

42 """ 

43 

44 def __init__(self, **kwargs): 

45 

46 # Initialise the OpenURLRequest object with empty attributes 

47 for key in JOURNAL_SCHEMA_KEYS: 

48 setattr(self, key, None) 

49 

50 # Save any attributes specified at creation time 

51 if kwargs: 

52 for key, value in kwargs.items(): 

53 setattr(self, key, value) 

54 

55 def __str__(self): 

56 return "OpenURLRequest{" + ", ".join(["%s : %s" % (x, getattr(self, x)) for x in JOURNAL_SCHEMA_KEYS if getattr(self, x)]) + "}"

58 def query_es(self): 

59 """ 

60 Query Elasticsearch for a set of matches for this request. 

61 :return: The results of a query through the dao, a JSON object. 

62 """ 

63 # Copy to the template, which will be populated with terms 

64 populated_query = deepcopy(TERMS_SEARCH) 

65 

66 # Get all of the attributes with values set. 

67 set_attributes = [(x, getattr(self, x)) for x in JOURNAL_SCHEMA_KEYS[:-1] if getattr(self, x)] 

68 

69 # If we don't have a genre, guess journal FIXME: is it correct to assume journal? 

70 if not self.genre: 

71 self.genre = SUPPORTED_GENRES[0] # TODO: we may want to handle 404 instead 

72 

73 # Set i to use either our mapping for journals or articles 

74 i = SUPPORTED_GENRES.index(getattr(self, 'genre').lower()) 

75 

76 # Add the attributes to the query 

77 for (k, v) in set_attributes: 

78 es_term = OPENURL_TO_ES[k][i] 

79 if es_term is None: 

80 continue 

81 else: 

82 term = {"term": {es_term: v}} 

83 populated_query["query"]["bool"]["must"].append(term) 

84 

85 # avoid doing an empty query 

86 if len(populated_query["query"]["bool"]["must"]) == 0: 

87 app.logger.debug("No valid search terms in OpenURL object") 

88 return None 

89 

90 # Return the results of the query 

91 if i == 0: 

92 app.logger.debug("OpenURL query to journal: " + json.dumps(populated_query)) 

93 return Journal.query(q=populated_query) 

94 elif i == 1: 

95 app.logger.debug("OpenURL query to article: " + json.dumps(populated_query)) 

96 return Article.query(q=populated_query)

98 def get_result_url(self): 

99 """ 

100 Get the URL for this OpenURLRequest's referent. 

101 :return: The url as a string, or None if not found. 

102 """ 

103 try: 

104 results = self.query_es() 

105 except ValueError: 

106 return None 

107 

108 if results is None: 

109 return None 

110 

111 if results.get('hits', {}).get('total', {}).get('value', 0) == 0: 

112 # No results found for query, retry 

113 results = self.fallthrough_retry() 

114 if results is None or results.get('hits', {}).get('total', {}).get('value', 0) == 0: 

115 # This time we've definitely failed 

116 return None 

117 

118 if results.get('hits', {}).get('hits', [{}])[0].get('_source', {}).get('es_type') == 'journal': 

119 

120 # construct a journal object around the result 

121 journal = Journal(**results['hits']['hits'][0]) 

122 

123 # the continuation is a first-class journal object, so if we have a journal we have the right continuation 

124 # (assuming that the user gave us specific enough information 

125 ident = journal.id 

126 

127 # If there request has a volume parameter, query for presence of an article with that volume 

128 if self.volume: 

129 vol_iss_results = self.query_for_vol(journal) 

130 

131 if vol_iss_results == None: 

132 # we were asked for a vol/issue, but weren't given the correct information to get it. 

133 return None 

134 elif vol_iss_results['hits']['total'] > 0: 

135 # construct the toc url using the ident, plus volume and issue 

136 jtoc_url = url_for("doaj.toc", identifier=ident, volume=self.volume, issue=self.issue) 

137 else: 

138 # If no results, the DOAJ does not contain the vol/issue being searched. (Show openurl 404) 

139 jtoc_url = None 

140 else: 

141 # if no volume parameter, construct the toc url using the ident only 

142 jtoc_url = url_for("doaj.toc", identifier=ident) 

143 return jtoc_url 

144 

145 elif results.get('hits', {}).get('hits', [{}])[0].get('_source', {}).get('es_type') == 'article': 

146 return url_for("doaj.article_page", identifier=results['hits']['hits'][0]['_id'])

148 def query_for_vol(self, journalobj): 

149 

150 # The journal object will already be the correct continuation, if the user provided sufficient detail. 

151 issns = journalobj.bibjson().issns() 

152 

153 # If there's no way to get the wanted issns, give up, else run the query 

154 if issns is None: 

155 return None 

156 else: 

157 volume_query = deepcopy(TERMS_SEARCH) 

158 volume_query["size"] = 0 

159 

160 issn_term = {"terms": {"index.issn.exact": issns}} 

161 volume_query["query"]["bool"]["must"].append(issn_term) 

162 

163 vol_term = {"term": {"bibjson.journal.volume.exact": self.volume}} 

164 volume_query["query"]["bool"]["must"].append(vol_term) 

165 

166 # And if there's an issue, query that too. Note, issue does not make sense on its own. 

167 if self.issue: 

168 iss_term = {"term": {"bibjson.journal.number.exact": self.issue}} 

169 volume_query["query"]["bool"]["must"].append(iss_term) 

170 

171 app.logger.debug("OpenURL subsequent volume query to article: " + json.dumps(volume_query)) 

172 return Article.query(q=volume_query)

174 def fallthrough_retry(self): 

175 """ 

176 Some things to try differently if we get no results on first attempt 

177 :return: a new result set, or None 

178 """ 

179 results = None 

180 

181 # Search again for the title against alternative_title (may catch translations of titles) 

182 if self.jtitle and not self.stitle: 

183 self.stitle = self.jtitle 

184 self.jtitle = None 

185 results = self.query_es() 

186 

187 return results

189 def validate_issn(self, issn_str): 

190 """ 

191 If the ISSN is missing a dash, add it so it matches that in the index. 

192 :param issn_str: The ISSN, or if None, this will skip. 

193 :return: The ISSN with the dash added 

194 """ 

195 if issn_str: 

196 match_dash = re.compile('[-]') 

197 if not match_dash.search(issn_str): 

198 issn_str = issn_str[:4] + '-' + issn_str[4:] 

199 return issn_str
